import pandas as pd
data={'ID':['U101','U102','U103','U104'],
      'Name':['A','B','C','D'],
      'Course':['BSc','BCom','BA','BSc'],
     }
df=pd.DataFrame(data)
print("Created Data Frame is:")
print(df)
print("\nAccessing the Columns:")
print("\nSingle Column:\n",df['Name'])
print("\nMulti Columns:\n",df[['Name','Course']])
print("\nAccessing the Rows:")
print("\nSingle Row:\n",df.loc[1])
print("\nMulti Rows:\n",df.loc[[1,3]])
print("\nFiltering the data:\n")
print(df[df['Course']=='BSc'])
print("\nAdding Marks column:\n")
df['Marks']=[85,90,70,95]
print(df)
print("\nDeleting the Marks column:\n")
del df['Marks']
